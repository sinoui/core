Checkbox 复选框组件。

import { useCallback } from 'react';
import Checkbox from '@sinoui/core/Checkbox';
import { ThemeProvider } from 'styled-components';
import { defaultTheme } from '@sinoui/theme';
import CheckboxDemo from './checkbox-demos/CheckboxDemo';
import CheckboxGroupDemo from './checkbox-demos/CheckboxGroupDemo';
import Demo from '../commons/Demo';

## Checkbox

```tsx
import Checkbox from '@sinoui/core/Checkbox';

function CheckboxDemo() {
  const [checked, setChecked] = useState(false);

  return (
    <Checkbox checked={checked} onChange={setChecked}>
      选项1
    </Checkbox>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <Demo>
    <CheckboxDemo>选项1</CheckboxDemo>
  </Demo>
</ThemeProvider>

## 状态

通过 `checked` 属性可以控制复选框的选中状态。`indeterminate` 属性用来指示复选框处于半选中状态。

```tsx
import Checkbox from '@sinoui/core/Checkbox';

function CheckboxDemo() {
  return (
    <>
      <Checkbox />
      <Checkbox checked />
      <Checkbox indeterminate />
    </>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <Demo>
    <Checkbox />
    <Checkbox checked />
    <Checkbox indeterminate />
  </Demo>
</ThemeProvider>

## 复选框分组

可以使用 `CheckboxGroup` 组件对复选框进行分组，如下所示：

```tsx
import CheckboxGorup from '@sinoui/core/CheckboxGroup';
import Checkbox from '@sinoui/core/Checkbox';

function CheckboxGroupDemo() {
  const [value, setValue] = useState(['2', '3']);
  return (
    <CheckboxGroup value={value} onChange={setValue}>
      <Checkbox value="1">选项1</Checkbox>
      <Checkbox value="2">选项2</Checkbox>
      <Checkbox value="3">选项3</Checkbox>
      <Checkbox value="4">选项4</Checkbox>
    </CheckboxGroup>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <Demo>
    <CheckboxGroupDemo />
  </Demo>
</ThemeProvider>

### 使用 items 指定一组复选框

`CheckboxGroup`组件支持使用`items`属性而非`children`指定一组复选框，如下所示：

```tsx
<CheckboxGroupDemo
  items={[
    { value: '1', label: '选项1' },
    { value: '2', label: '选项2' },
    { value: '3', label: '选项3' },
    { value: '4', label: '选项4' },
  ]}
/>
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <Demo>
    <CheckboxGroupDemo />
  </Demo>
</ThemeProvider>

### 分组布局

复选框分组默认按照水平换行的方式布局复选框。`CheckboxGorup`组件支持以列的方式布局，如下所示：

```tsx
<CheckboxGroup column>
  <Checkbox value="1">选项1</Checkbox>
  <Checkbox value="2">选项2</Checkbox>
  <Checkbox value="3">选项3</Checkbox>
  <Checkbox value="4">选项4</Checkbox>
</CheckboxGroup>
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <Demo>
    <CheckboxGroupDemo column />
  </Demo>
</ThemeProvider>

也可以通过`columns`属性指定列数，如下所示：

```tsx
<CheckboxGroup columns={2}>
  <Checkbox value="1">选项1</Checkbox>
  <Checkbox value="2">选项2</Checkbox>
  <Checkbox value="3">选项3</Checkbox>
  <Checkbox value="4">选项4</Checkbox>
</CheckboxGroup>
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <Demo>
    <CheckboxGroupDemo columns={2} />
  </Demo>
</ThemeProvider>

### 全选

`CheckboxGroup`组件有`enableSelectAll`属性，支持全选功能：

```tsx
<CheckboxGroup enableSelectAll>
  <Checkbox value="1">选项1</Checkbox>
  <Checkbox value="2">选项2</Checkbox>
  <Checkbox value="3">选项3</Checkbox>
  <Checkbox value="4">选项4</Checkbox>
</CheckboxGroup>
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <Demo>
    <CheckboxGroupDemo enableSelectAll />
  </Demo>
</ThemeProvider>

## 不可用、只读

```tsx
import Checkbox from '@sinoui/core/Checkbox';

function CheckboxDemo() {
  return (
    <>
      不可用：
      <Checkbox disabled />
      只读：
      <Checkbox readOnly />
      分组不可用：
      <CheckboxGroup disabled>
        <Checkbox value="1">选项1</Checkbox>
        <Checkbox value="2">选项2</Checkbox>
        <Checkbox value="3">选项3</Checkbox>
        <Checkbox value="4">选项4</Checkbox>
      </CheckboxGroup>
      分组只读：
      <CheckboxGroup readOnly>
        <Checkbox value="1">选项1</Checkbox>
        <Checkbox value="2">选项2</Checkbox>
        <Checkbox value="3">选项3</Checkbox>
        <Checkbox value="4">选项4</Checkbox>
      </CheckboxGroup>
    </>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <Demo style={{ flexDirection: 'column' }}>
    <span>不可用：</span>
    <Checkbox disabled />
    <span>只读：</span>
    <Checkbox readOnly />
    <span>分组不可用：</span>
    <CheckboxGroupDemo disabled />
    <span>分组只读：</span>
    <CheckboxGroupDemo readOnly />
  </Demo>
</ThemeProvider>
