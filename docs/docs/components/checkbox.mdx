Checkbox 复选框组件。

import { useCallback } from 'react';
import Checkbox from '@sinoui/core/Checkbox';
import { ThemeProvider } from 'styled-components';
import { defaultTheme } from '@sinoui/theme';
import CheckboxDemo from './checkbox-demos/CheckboxDemo';
import CheckboxGroupDemo from './checkbox-demos/CheckboxGroupDemo';

## Checkbox

```tsx
import Checkbox from '@sinoui/core/Checkbox';

function CheckboxDemo() {
  const [checked, setChecked] = useState(false);

  return (
    <Checkbox checked={checked} onChange={setChecked}>
      选项1
    </Checkbox>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <CheckboxDemo>选项1</CheckboxDemo>
</ThemeProvider>

## 状态

通过 `checked` 属性可以控制复选框的选中状态。`indeterminate` 属性用来指示复选框处于半选中状态。

```tsx
import Checkbox from '@sinoui/core/Checkbox';

function CheckboxDemo() {
  return (
    <>
      <Checkbox />
      <Checkbox checked />
      <Checkbox indeterminate />
    </>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <>
    <Checkbox />
    <Checkbox checked />
    <Checkbox indeterminate />
  </>
</ThemeProvider>

## 复选框分组

可以使用 `CheckboxGroup` 组件对复选框进行分组，如下所示：

```tsx
import CheckboxGorup from '@sinoui/core/CheckboxGroup';
import Checkbox from '@sinoui/core/Checkbox';

function CheckboxGroupDemo() {
  return (
    <CheckboxGroup>
      <Checkbox value="1">选项1</Checkbox>
      <Checkbox value="2">选项2</Checkbox>
      <Checkbox value="3">选项3</Checkbox>
      <Checkbox value="4">选项4</Checkbox>
    </CheckboxGroup>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <CheckboxGroupDemo />
</ThemeProvider>

## 设置只读

```tsx
import Checkbox from '@sinoui/core/Checkbox';

function CheckboxDemo() {
  return (
    <>
      不可用：
      <Checkbox disabled />
      只读：
      <Checkbox readOnly />
    </>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <>
    不可用：
    <Checkbox disabled />
    只读：
    <Checkbox readOnly />
  </>
</ThemeProvider>

## 指定颜色

```tsx
import Checkbox from '@sinoui/core/Checkbox';

function CheckboxDemo() {
  return (
    <>
      <Checkbox checked color="primary" />
      <Checkbox checked color="secondary" />
      <Checkbox checked color="error" />
      <Checkbox checked color="warning" />
      <Checkbox checked color="success" />
      <Checkbox checked color="info" />
      <Checkbox checked color="actionActive" />
      <Checkbox checked color="actionDisabled" />
    </>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <>
    <Checkbox checked color="primary" />
    <Checkbox checked color="secondary" />
    <Checkbox checked color="error" />
    <Checkbox checked color="warning" />
    <Checkbox checked color="success" />
    <Checkbox checked color="info" />
    <Checkbox checked color="actionActive" />
    <Checkbox checked color="actionDisabled" />
  </>
</ThemeProvider>
