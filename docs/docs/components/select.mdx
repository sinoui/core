> 选择框组件。能从一个选项列表中去获得用户所提供的信息。

## 引入方式

```ts
import Select from '@sinoui/core/Select';
```

## 基本使用

```tsx
import React, { useState } from 'react';
import Select, { Option } from '@sinoui/core/Select';
import styled, { ThemeProvider } from 'styled-components';
import { defaultTheme } from '@sinoui/theme';

const SelectField = styled(Select)`
  margin: 8px;
  width: 200px;
`;

export default function SimpleDemo() {
  const [value, setValue] = useState('');

  const onChange = (val: any) => {
    setValue(val);
  };
  return (
    <ThemeProvider theme={defaultTheme}>
      <div
        style={{
          display: 'flex',
          alignItems: 'flex-start',
          flexWrap: 'wrap',
        }}
      >
        <SelectField value={value} onChange={onChange} label="Select">
          <Option value="选项一">选项一</Option>
          <Option value="选项二">选项二</Option>
          <Option value="选项三">选项三</Option>
          <Option value="选项四">选项四</Option>
        </SelectField>
        <SelectField
          value={value}
          onChange={onChange}
          label="Select"
          variant="filled"
        >
          <Option value="选项一">选项一</Option>
          <Option value="选项二">选项二</Option>
          <Option value="选项三">选项三</Option>
          <Option value="选项四">选项四</Option>
        </SelectField>
        <SelectField
          value={value}
          onChange={onChange}
          label="Select"
          variant="outlined"
        >
          <Option value="选项一">选项一</Option>
          <Option value="选项二">选项二</Option>
          <Option value="选项三">选项三</Option>
          <Option value="选项四">选项四</Option>
        </SelectField>
      </div>
    </ThemeProvider>
  );
}
```

## 不同状态的选择框

`Select` 支持基本的表单属性，包括：

- `required` - 必填
- `disabled` - 禁用
- `readOnly` - 只读

NativeSelect 组件还支持一些常用的属性，如：

- `helperText` - 帮助性文本
- `error` - 选择框的错误信息

更多关于 Select 的属性，请参考[Select API 文档](../components-apis/select)。

## 多重选择

我们可以将`Select`的`multiple`属性设置为`true`，从而支持多重选择。

```tsx
import React, { useState } from 'react';
import Select, { Option } from '@sinoui/core/Select';
import styled, { ThemeProvider } from 'styled-components';
import { defaultTheme } from '@sinoui/theme';

const SelectField = styled(Select)`
  margin: 8px;
  width: 200px;
`;

export default function MultipleSelectDemo() {
  const [value, setValue] = useState([]);

  const onChange = (val: any) => {
    setValue(val);
  };
  return (
    <ThemeProvider theme={defaultTheme}>
      <div
        style={{
          display: 'flex',
          alignItems: 'flex-start',
          flexWrap: 'wrap',
        }}
      >
        <SelectField
          value={value}
          onChange={onChange}
          label="Multiple Select"
          multiple
        >
          <Option value="选项一">选项一</Option>
          <Option value="选项二">选项二</Option>
          <Option value="选项三">选项三</Option>
          <Option value="选项四">选项四</Option>
        </SelectField>
        <SelectField
          value={value}
          onChange={onChange}
          label="Multiple Select"
          multiple
          variant="filled"
        >
          <Option value="选项一">选项一</Option>
          <Option value="选项二">选项二</Option>
          <Option value="选项三">选项三</Option>
          <Option value="选项四">选项四</Option>
        </SelectField>
        <SelectField
          value={value}
          onChange={onChange}
          label="Multiple Select"
          multiple
          variant="outlined"
        >
          <Option value="选项一">选项一</Option>
          <Option value="选项二">选项二</Option>
          <Option value="选项三">选项三</Option>
          <Option value="选项四">选项四</Option>
        </SelectField>
      </div>
    </ThemeProvider>
  );
}
```
