import Fab from '@sinoui/core/Fab';
import { MdAdd } from "react-icons/md";
import StoryLayout from './StoryLayout';

## 常规 Fab

```tsx
import Fab from '@sinoui/core/Fab';

function FabDemo() {
  return (
    <Fab>
      <MdAdd />
    </Fab>
  );
}
```

效果如下：

<StoryLayout>
  <Fab>
    <MdAdd />
  </Fab>
</StoryLayout>
