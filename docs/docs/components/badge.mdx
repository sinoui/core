徽章会在其子项的右上角生成一个小徽章。

import { ThemeProvider } from 'styled-components';
import { defaultTheme } from '@sinoui/theme';
import SvgIcon from '@sinoui/core/SvgIcon';
import { MdEmail } from 'react-icons/md';
import Badge from '@sinoui/core/Badge';

## 基本使用

```tsx
import Badge from '@sinoui/core/Badge';

function BadgeDemo() {
  return (
    <Badge>
      <SvgIcon as={MdEmail} />
    </Badge>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <Badge>
    <SvgIcon as={MdEmail} />
  </Badge>
</ThemeProvider>

## 指定封顶数值

```tsx
import Badge from '@sinoui/core/Badge';

function BadgeDemo() {
  return (
    <>
      <Badge count={99} overflowCount={99}>
        <SvgIcon as={MdEmail} />
      </Badge>
      <p />
      <Badge count={100} overflowCount={99}>
        <SvgIcon as={MdEmail} />
      </Badge>
    </>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <Badge count={99} overflowCount={99}>
    <SvgIcon as={MdEmail} />
  </Badge>
  <p />
  <Badge count={100} overflowCount={99}>
    <SvgIcon as={MdEmail} />
  </Badge> 
</ThemeProvider>

## 设置徽标的显示位置

    <>
      <Badge count={8} anchorOrigin={{ vertical: 'top', horizontal: 'right' }}>
        <SvgIcon as={MdEmail} />
      </Badge>
      <p />
      <Badge count={8} anchorOrigin={{ vertical: 'top', horizontal: 'left' }}>
        <SvgIcon as={MdEmail} />

      <p />
      <Badge count={8} anchorOrigin={{ vertical: 'bottom', horizontal: 'right' }}>

      <Badge count={8} anchorOrigin={{ vertical: 'bottom', horizontal: 'left' }}>
        <SvgIcon as={MdEmail} />
