> 日期选择允许用户选择日期或日期范围，时间选择允许用户选择时间。

## 引用方式

```ts
import DatePicker from '@sinoui/core/DatePicker'; // 日期选择
import DateRangePicker from '@sinoui/core/DateRangePicker'; // 日期区间选择
import TimePicker from '@sinoui/core/TimePicker'; // 时间选择
```

## 日期选择

### 基本使用

```tsx
function DatePickerDemo(props: any) {
  const [value, setValue] = useState<string | undefined>('2020-06-12');
  return (
    <Datepicker
      value={value}
      onChange={(_value) => setValue(_value)}
      {...props}
    />
  );
}
```

<DatePickerDemo />

### 设置最大最小值

在使用 DatePicker 时，我们可以通过`min`和`max`分别设置可选的最小范围和最大范围。使用方式如下:

```tsx
import React, { useState } from 'react';
import DatePicker from '@sinoui/core/DatePicker';
import styled, { ThemeProvider } from 'styled-components';
import { defaultTheme } from '@sinoui/theme';

const StyledDatePicker = styled(DatePicker)`
  width: 200px;
`;

export default function DatePickerDemo(props: any) {
  const [value, setValue] = useState<string | undefined>('2020-06-12');
  return (
    <ThemeProvider theme={defaultTheme}>
      <StyledDatePicker
        value={value}
        onChange={(_value) => setValue(_value)}
        {...props}
        min="2020-06-09"
        max="2020-06-24"
      />
    </ThemeProvider>
  );
}
```

<DatePickerDemo min="2020-06-09" max="2020-06-24" />

### 自定义值的渲染

DatePicker 组件默认直接显示 value 的值。如果我们想要调整其显示方式，可以使用`renderValue`属性来自定义渲染方式，如`2020-01-01`:

```tsx
function DatePickerDemo(props: any) {
  const [value, setValue] = useState<string | undefined>('2020-06-12');
  return (
    <Datepicker
      value={value}
      onChange={(_value) => setValue(_value)}
      renderValue={(_value) =>
        _value ? dayjs(_value).format('YYYY-MM-DD') : ''
      }
      {...props}
    />
  );
}
```

<DatePickerDemo
  renderValue={(_value) => (_value ? dayjs(_value).format('YYYY-MM-DD') : '')}
/>

## 日期区间选择(DateRangePicker)

```tsx
import React, { useState } from 'react';
import DateRangePicker from '@sinoui/core/DateRangePicker';
import styled, { ThemeProvider } from 'styled-components';
import { defaultTheme } from '@sinoui/theme';

const StyledDateRangePicker = styled(DateRangePicker)`
  width: 140px;
`;

export default function DateRangePickerDemo() {
  const [value, setValue] = useState(['2020-06-16', '2020-07-28']);
  return (
    <ThemeProvider theme={defaultTheme}>
      <StyledDateRangePicker value={value} onChange={setValue} />
    </ThemeProvider>
  );
}
```

<DateRangePickerDemo />

## 时间选择(TimePicker)

### 基本用法

```tsx
import React, { useState } from 'react';
import TimePicker from '@sinoui/core/TimePicker';
import { ThemeProvider } from 'styled-components';
import { defaultTheme } from '@sinoui/theme';

export default function DatePickerDemo(props: any) {
  const [value, setValue] = useState<string | undefined>('14:20');
  return (
    <ThemeProvider theme={defaultTheme}>
      <TimePicker
        value={value}
        onChange={(_value) => setValue(_value)}
        {...props}
      />
    </ThemeProvider>
  );
}
```
