CheckboxGroup 复选框组组件。

import { useCallback } from 'react';
import CheckboxGroup from '@sinoui/core/CheckboxGroup';
import Checkbox from '@sinoui/core/Checkbox';
import { ThemeProvider } from 'styled-components';
import { defaultTheme } from '@sinoui/theme';
import CheckboxGroupDemo from './checkbox-demos/CheckboxGroupDemo';

## CheckboxGroup

## 基本使用

```tsx
import CheckboxGroup from '@sinoui/core/CheckboxGroup';

function CheckboxGroupDemo() {
  const [value, setValue] = useState(['2', '3']);

  return (
    <CheckboxGroup onChange={setValue} value={value}>
      <Checkbox value="1">选项1</Checkbox>
      <Checkbox value="2">选项2</Checkbox>
      <Checkbox value="3">选项3</Checkbox>
      <Checkbox value="4">选项4</Checkbox>
    </CheckboxGroup>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <CheckboxGroupDemo />
</ThemeProvider>

## 设置 items 属性

```tsx
import CheckboxGroup from '@sinoui/core/CheckboxGroup';

function CheckboxGroupDemo() {
  const [value, setValue] = useState(['2', '3']);

  return (
    <CheckboxGroup
      onChange={setValue}
      value={value}
      items={[
        { value: '1', label: '选项1' },
        { value: '2', label: '选项2' },
        { value: '3', label: '选项3' },
        { value: '4', label: '选项4' },
      ]}
    />
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <CheckboxGroupDemo
    items={[
      { value: '1', label: '选项1' },
      { value: '2', label: '选项2' },
      { value: '3', label: '选项3' },
      { value: '4', label: '选项4' },
    ]}
  />
</ThemeProvider>

## 设置不可用

```tsx
import CheckboxGroup from '@sinoui/core/CheckboxGroup';

function CheckboxGroupDemo() {
  return (
    <>
      不可用:
      <CheckboxGroup disabled value={['2']}>
        <Checkbox value="1">选项1</Checkbox>
        <Checkbox value="2">选项2</Checkbox>
      </CheckboxGroup>
      <br />
      只读:
      <CheckboxGroup readOnly value={['2']}>
        <Checkbox value="1">选项1</Checkbox>
        <Checkbox value="2">选项2</Checkbox>
      </CheckboxGroup>
    </>

}
```

## 设置标题显示位置

```tsx
import CheckboxGroup from '@sinoui/core/CheckboxGroup';

function CheckboxGroupDemo() {
  const [value, setValue] = useState(['2', '3']);

  return (
    <>
      标题显示在右侧：
      <CheckboxGroup onChange={setValue} value={value}>
        <Checkbox value="1">选项1</Checkbox>
        <Checkbox value="2">选项2</Checkbox>
      </CheckboxGroup>
      <br />
      标题显示在左侧：
      <CheckboxGroup labelPosition="left" onChange={setValue} value={value}>
        <Checkbox value="1">选项1</Checkbox>
        <Checkbox value="2">选项2</Checkbox>
      </CheckboxGroup>
    </>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <>
    标题显示在右侧：
    <CheckboxGroupDemo />
    <br />
    标题显示在左侧：
    <CheckboxGroupDemo labelPosition="left" />
  </>
</ThemeProvider>

## 设置网格对齐布局

可以通过 `columns` 属性指定网格布局的列数，如下所示：

```tsx
import CheckboxGroup from '@sinoui/core/CheckboxGroup';

function CheckboxGroupDemo() {
  const [value, setValue] = useState(['2', '3']);

  return (
    <CheckboxGroup columns={3} onChange={setValue} value={value}>
      <Checkbox value="1">选项1</Checkbox>
      <Checkbox value="2">选项2</Checkbox>
      <Checkbox value="3">选项3</Checkbox>
      <Checkbox value="4">选项4</Checkbox>
      <Checkbox value="5">选项5</Checkbox>
    </CheckboxGroup>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <CheckboxGroupDemo columns={3} />
</ThemeProvider>
