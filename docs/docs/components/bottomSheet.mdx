## 基本使用

```tsx
import BottomSheet from '@sinoui/core/BottomSheet';

function BottomSheetDemo() {
  const listData = [1, 2, 3];
  const content = (
    <List>
      {listData.map((item) => (
        <>
          <ListItem>
            <ListItemPrimaryAction>
              <IconButton color="primary">
                <BookmarkBorder />
              </IconButton>
            </ListItemPrimaryAction>
            <ListItemText>item{item}</ListItemText>
          </ListItem>
          <Divider />
        </>
      ))}
    </List>
  );

  return <BottomSheet>{content}</BottomSheet>
}
```

## 点击遮罩层事件

```tsx
import BottomSheet from '@sinoui/core/BottomSheet';

function BottomSheetDemo() {
  const [open, setOpen] = useState(false);

  const onClick = () => {
    setOpen(!open);
  };

  const onBackdropClick = () => {
    setOpen(false);
  };

  const listData = [1, 2, 3];

  const content = (
    <List>
      {listData.map((item) => (
        <>
          <ListItem>
            <ListItemPrimaryAction>
              <IconButton color="primary">
                <BookmarkBorder />
              </IconButton>
            </ListItemPrimaryAction>
            <ListItemText>item{item}</ListItemText>
          </ListItem>
          <Divider />
        </>
      ))}
    </List>
  );

  return (
    <>
      <Button onClick={onClick}>CLICK</Button>
      <BottomSheet open={open} onBackdropClick={onBackdropClick}>
        {content}
      </BottomSheet>
    </>
  );
}
```

## 设置不显示遮罩层

```tsx
import BottomSheet from '@sinoui/core/BottomSheet';

function BottomSheetDemo() {
  const [open, setOpen] = useState(false);

  const onClick = () => {
    setOpen(!open);
  };

  const listData = [1, 2, 3];

  const content = (
    <List>
      {listData.map((item) => (
        <>
          <ListItem>
            <ListItemPrimaryAction>
              <IconButton color="primary">
                <BookmarkBorder />
              </IconButton>
            </ListItemPrimaryAction>
            <ListItemText>item{item}</ListItemText>
          </ListItem>
          <Divider />
        </>
      ))}
    </List>
  );

  return (
    <>
      <Button onClick={onClick}>CLICK</Button>
      <BottomSheet open={open} backdrop={false}>
        {content}
      </BottomSheet>
    </>
  );
}
```
