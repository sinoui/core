## Radio

```tsx
import React, { useState } from 'react';
import Radio from '@sinoui/core/Radio';

export default function RadioDemo() {
  const [checked, setChecked] = useState(false);

  return <Radio checked={checked} onChange={setChecked} />;
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <RadioDemo>选项1</RadioDemo>
</ThemeProvider>

## 状态

通过 `checked` 属性可以控制单选按钮的选中状态。

```tsx
import Radio from '@sinoui/core/Radio';

function RadioDemo() {
  return (
    <>
      <Radio />
      <Radio checked />
    </>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>

    <Radio />
    <Radio checked />
  </>
</ThemeProvider>

## RadioGroup

## 基本使用

```tsx
import RadioGroup from '@sinoui/core/RadioGroup';

function RadioGroupDemo() {
  const [value, setValue] = useState('2');

  const onChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    setValue(event.target.value);
  };

  return (
    <RadioGroup onChange={onChange} value={value}>
      <Radio value="1">单选框1</Radio>
      <Radio value="2">单选框2</Radio>
      <Radio value="3">单选框3</Radio>
      <Radio value="4">单选框4</Radio>
    </RadioGroup>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <RadioGroupDemo />
</ThemeProvider>

        <Radio value="3">单选框3</Radio>

## 设置标题显示位置

```tsx
import RadioGroup from '@sinoui/core/RadioGroup';

function RadioGroupDemo() {
  const [value, setValue] = useState('2');

  const onChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    setValue(event.target.value);
  };

  return (
    <>
      标题在右侧显示：
      <RadioGroup onChange={onChange} value={value}>
        <Radio value="1">单选框1</Radio>
        <Radio value="2">单选框2</Radio>
        <Radio value="3">单选框3</Radio>
        <Radio value="4">单选框4</Radio>
      </RadioGroup>
      <br />
      标题在左侧显示：
      <RadioGroup labelPosition="left" onChange={onChange} value={value}>
        <Radio value="1">单选框1</Radio>
        <Radio value="2">单选框2</Radio>
        <Radio value="3">单选框3</Radio>
        <Radio value="4">单选框4</Radio>
      </RadioGroup>
    </>
  );
}
```

效果如下：

<ThemeProvider theme={defaultTheme}>
  <>
    标题在右侧显示：
    <RadioGroupDemo />
    <br />
    标题在左侧显示：
    <RadioGroupDemo labelPosition="left" />
  </>
</ThemeProvider>
